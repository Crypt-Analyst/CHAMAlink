üíª Quick Start Example

# Python Example
import requests

# Authentication
headers = {
    'Authorization': 'Bearer YOUR_API_TOKEN',
    'Content-Type': 'application/json'
}

# Get all chamas
response = requests.get(
    'https://api.chamalink.com/v1/chamas',
    headers=headers
)

chamas = response.json()
print(f"Found {len(chamas)} chamas")

# Create a new transaction
transaction_data = {
    'chama_id': 'chama_123',
    'amount': 5000,
    'type': 'contribution',
    'description': 'Monthly contribution',
    'member_id': 'member_456'
}

response = requests.post(
    'https://api.chamalink.com/v1/transactions',
    headers=headers,
    json=transaction_data
)

print(f"Transaction created: {response.json()}")
